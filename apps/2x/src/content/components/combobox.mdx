## Usage

```tsx title="Imports"
import {
	Combobox,
	ComboboxChip,
	ComboboxChips,
	ComboboxContent,
	ComboboxEmpty,
	ComboboxInput,
	ComboboxItem,
	ComboboxList,
	ComboboxTrigger,
	ComboboxValue,
} from "@/components/ui/combobox"
```

```tsx title="Anatomy"
<Combobox>
	<ComboboxInput />
	<ComboboxTrigger />
	<ComboboxValue />
	<ComboboxChips>
		<ComboboxChip />
	</ComboboxChips>
	<ComboboxContent>
		<ComboboxEmpty />
		<ComboboxList>
			<ComboboxItem />
		</ComboboxList>
	</ComboboxContent>
</Combobox>
```
