## Usage

```tsx title="Imports"
import {
	Carousel,
	CarouselContent,
	CarouselItem,
	CarouselNext,
	CarouselPrevious,
} from "@/components/ui/carousel"
```

```tsx title="Anatomy"
<Carousel>
	<CarouselContent>
		<CarouselItem />
	</CarouselContent>
	<CarouselPrevious />
	<CarouselNext />
</Carousel>
```
