## Usage

```tsx title="Imports"
import {
	Accordion,
	AccordionContent,
	AccordionItem,
	AccordionTrigger,
} from "@/components/ui/autocomplete"
```

```tsx title="Anatomy"
<Autocomplete>
	<AutocompleteItem>
		<AccordionTrigger />
		<AccordionContent />
	</AutocompleteItem>
</Autocomplete>
```
